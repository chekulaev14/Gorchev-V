# Routing 2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a new `Маршруты 2` warehouse tab with a visual constructor that stays local until `Записать`, then publishes one chain into real `Item` and `Routing` records.

**Architecture:** Reuse the existing warehouse shell and item/process sources, but introduce a separate `routing-v2` publish boundary. The new UI owns constructor state locally and submits one typed payload to a dedicated route/service, which creates missing items and one routing in a single transaction.

**Tech Stack:** Next.js App Router, React, TypeScript, Zod, Prisma, Vitest

- [ ] **Step 2: Create the schema**

Create `app/src/lib/schemas/routing-v2.schema.ts`:

```ts
import { z } from "zod";

const draftItemSchema = z.object({
  name: z.string().min(1),
  type: z.enum(["material", "blank", "product"]),
  unit: z.enum(["kg", "pcs", "m"]),
  side: z.enum(["LEFT", "RIGHT", "NONE"]),
  pricePerUnit: z.number().nonnegative().optional(),
});

const nodeSchema = z.discriminatedUnion("source", [
  z.object({
    id: z.string().min(1),
    source: z.literal("existing"),
    itemId: z.string().min(1),
  }),
  z.object({
    id: z.string().min(1),
    source: z.literal("new"),
    draftItem: draftItemSchema,
  }),
]);

const edgeSchema = z.object({
  fromNodeId: z.string().min(1),
  toNodeId: z.string().min(1),
  qty: z.number().positive(),
  sortOrder: z.number().int().min(0),
});

const stepSchema = z.object({
  stepNo: z.number().int().min(1),
  outputNodeId: z.string().min(1),
  outputQty: z.number().positive(),
  processId: z.string().min(1),
});

export const publishRoutingV2Schema = z.object({
  chainName: z.string().min(1),
  productNodeId: z.string().min(1),
  nodes: z.array(nodeSchema).min(1),
  edges: z.array(edgeSchema),
  steps: z.array(stepSchema).min(1),
});

export type PublishRoutingV2Input = z.infer<typeof publishRoutingV2Schema>;
```

- [ ] **Step 2: Implement the publish service**

Create `app/src/services/routing-v2-publish.service.ts`:

```ts
import { prisma } from "@/lib/prisma";
import { ServiceError } from "@/lib/api/handle-route-error";
import { getNextCode, toCodeKind } from "@/services/helpers/code-generator";
import type { PublishRoutingV2Input } from "@/lib/schemas/routing-v2.schema";

export async function publishRoutingV2(input: PublishRoutingV2Input) {
  return prisma.$transaction(async (tx) => {
    const nodeItemMap = new Map<string, string>();
    const createdItems: { nodeId: string; itemId: string }[] = [];

    for (const node of input.nodes) {
      if (node.source === "existing") {
        const item = await tx.item.findUnique({ where: { id: node.itemId } });
        if (!item) throw new ServiceError("Номенклатура не найдена", 404);
        nodeItemMap.set(node.id, item.id);
        continue;
      }

      const id = crypto.randomUUID();
      const code = await getNextCode(tx, toCodeKind(node.draftItem.type));
      const created = await tx.item.create({
        data: {
          id,
          code,
          name: node.draftItem.name,
          typeId: node.draftItem.type,
          unitId: node.draftItem.unit,
          side: node.draftItem.side,
          pricePerUnit: node.draftItem.pricePerUnit ?? null,
          images: [],
        },
      });
      nodeItemMap.set(node.id, created.id);
      createdItems.push({ nodeId: node.id, itemId: created.id });
    }

    const productItemId = nodeItemMap.get(input.productNodeId);
    if (!productItemId) throw new ServiceError("Не найден конечный узел", 400);

    const maxVersion = await tx.routing.aggregate({
      where: { itemId: productItemId },
      _max: { version: true },
    });

    const routing = await tx.routing.create({
      data: {
        itemId: productItemId,
        version: (maxVersion._max.version ?? 0) + 1,
        status: "DRAFT",
        steps: {
          create: input.steps.map((step) => ({
            stepNo: step.stepNo,
            processId: step.processId,
            outputItemId: nodeItemMap.get(step.outputNodeId)!,
            outputQty: step.outputQty,
            inputs: {
              create: input.edges
                .filter((edge) => edge.toNodeId === step.outputNodeId)
                .map((edge) => ({
                  itemId: nodeItemMap.get(edge.fromNodeId)!,
                  qty: edge.qty,
                  sortOrder: edge.sortOrder,
                })),
            },
          })),
        },
      },
      include: { steps: { include: { inputs: true }, orderBy: { stepNo: "asc" } } },
    });

    return { routing, createdItems };
  });
}
```

import { NextResponse } from "next/server";
import { handleRouteError } from "@/lib/api/handle-route-error";
import { parseBody } from "@/lib/schemas/helpers";
import { publishRoutingV2Schema } from "@/lib/schemas/routing-v2.schema";
import { publishRoutingV2 } from "@/services/routing-v2-publish.service";

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const parsed = parseBody(publishRoutingV2Schema, body);
    if (!parsed.success) return parsed.response;

    const result = await publishRoutingV2(parsed.data);
    return NextResponse.json(result, { status: 201 });
  } catch (err) {
    return handleRouteError(err);
  }
}
```

import type { PublishRoutingV2Input } from "@/lib/schemas/routing-v2.schema";
import type { Routing2Edge, Routing2Node } from "./routing2-types";

export function buildPublishPayload(args: {
  chainName: string;
  productNodeId: string;
  nodes: Routing2Node[];
  edges: Routing2Edge[];
  steps: { stepNo: number; outputNodeId: string; outputQty: number; processId: string }[];
}): PublishRoutingV2Input {
  return {
    chainName: args.chainName,
    productNodeId: args.productNodeId,
    nodes: args.nodes.map((node) =>
      node.source === "existing"
        ? { id: node.id, source: "existing", itemId: node.itemId! }
        : { id: node.id, source: "new", draftItem: node.draftItem! },
    ),
    edges: args.edges,
    steps: args.steps,
  };
}
```
